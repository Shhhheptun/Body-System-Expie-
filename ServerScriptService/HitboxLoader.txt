local Players = game:GetService("Players")

local HitboxService = require(
	game.ServerScriptService.BodySystem.HitboxService
)

Players.PlayerAdded:Connect(function(player)

		task.wait(0.5)

		HitboxService.Attach(character)

	end)

end)
