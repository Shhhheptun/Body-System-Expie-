-- PainSystem
-- собирает боль с конечностей + реген

local PainSystem = {}

local function clamp(v, min, max)
	return math.max(min, math.min(max, v))
end

function PainSystem.Update(body, dt)

	-- ===== СБОР БОЛИ С КОНЕЧНОСТЕЙ =====
	local maxPain = 0

	for _, part in pairs(body.Parts) do
		local limbPain = part.LimbPain or 0

		if limbPain > maxPain then
			maxPain = limbPain
		end
	end

	-- ===== БАЗОВАЯ БОЛЬ =====
	body.BasePain = clamp(
		maxPain,
		body.MinPain or 0,
		body.MaxPain or 100
	)

	-- ===== РЕГЕН БОЛИ =====
	-- уменьшаем боль со временем (на конечностях!)
	local regenRate = 1 -- В СЕКУНДУ (крути эту хуню)

	for _, part in pairs(body.Parts) do
		if part.LimbPain and part.LimbPain > 0 then
			part.LimbPain = clamp(
				part.LimbPain - regenRate * dt,
				0,
				body.MaxPain or 100
			)
		end
	end
end

return PainSystem
